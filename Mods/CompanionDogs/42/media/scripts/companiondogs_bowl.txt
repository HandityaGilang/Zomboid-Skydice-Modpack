module Base
{
    /* Companion Dogs Bowl: a single placeable ITEM that IS the feeding station: drop it on the ground
       (IsoWorldInventoryObject) and it renders the pet WaterDish 3D model. It holds food OR water (one at a time, like
       the vanilla trough, the fill commands refuse the opposite kind until it's emptied); the auto-feed
       (CompanionDogs_Companion) reads how much from the item ModData (cdFoodMeals OR cdWater, a 0..100 budget).
       VISUAL (lives-updates, no chunk reload): the bowl is a cosmetic FluidContainer. CompanionDogs_Util.refreshDishModel
       drives the fluid to reflect the stock: empty = no fluid (grey dish), food = CompanionDogsKibble (brown), water =
       CompanionDogsBowlWater (blue). The engine then auto-renders the vanilla WaterDish_Fluid model tinted by the fluid
       COLOR over the FLUIDTINT octagon (WorldItemAtlas), and re-snapshots LIVE whenever the fluid level/color changes
       (isStillValid). This replaces an earlier setWorldStaticModel+worldScale hack that crashed (native render crash).
       The two cosmetic fluids are NOT water/beverage (Industrial), so the bowl is never a water source nor drinkable,
       and RainFactor 0 keeps the rain from changing the look; the fluid is purely visual, gameplay stays in ModData. */
    fluid CompanionDogsKibble
    {
        DisplayName = Fluid_Name_CompanionDogsKibble,
        ColorReference = Sienna,
        Categories
        {
            Industrial,
        }
    }

    fluid CompanionDogsBowlWater
    {
        DisplayName = Fluid_Name_CompanionDogsBowlWater,
        ColorReference = DeepSkyBlue,
        Categories
        {
            Industrial,
        }
    }

    item CompanionDogsBowl
    {
        DisplayCategory = Animal,
        ItemType = base:normal,
        Weight = 1.5,
        Icon = Waterdish_Grey,
        IconFluidMask = WaterDish_Mask,
        WorldStaticModel = WaterDish,
        Tags = base:OmitEmptyFromName,
        component FluidContainer
        {
            ContainerName = CompanionDogsBowl,
            Capacity = 0.3,
            RainFactor = 0,
            InitialPercentMin = 0,
            InitialPercentMax = 0,
            Fluids
            {
                fluid = CompanionDogsKibble:1.0,
                fluid = CompanionDogsBowlWater:1.0,
            }
        }
    }

    /* The bowl is obtained by repurposing ANY cooking vessel / dish (base:cookable = pots, pans, baking/roasting pans,
       kettles, buckets, mugs, bottles…), an easy, always-available recipe (no carpentry, no skill). It also spawns in
       loot (see CompanionDogs_Distributions.lua). The recipe's UI label comes from Translate/<lang>/Recipes.json. */
    craftRecipe MakeCompanionDogsBowlFromCookware
    {
        time = 60,
        timedAction = Making,
        category = Farming,
        Tags = InHandCraft;Farming,
        inputs
        {
            /* flags[ItemCount] = consumir 1 ITEM (nao 1 "uso"); sem ele o engine trata cookables com FluidContainer
               (panela, chaleira, balde) por semantica de fluido e o craft falha em silencio (padrao vanilla MakeJar) */
            item 1 tags[base:cookable] mode:destroy flags[ItemCount],
        }
        outputs
        {
            item 1 Base.CompanionDogsBowl,
        }
    }

    /* Same bowl from the vanilla pet Water Dish (Base.WaterDish) — the game's pet dish, same WaterDish model our bowl
       uses. A single input can't mix tags[base:cookable] with a specific item, so this is a sibling recipe; both output
       the identical Base.CompanionDogsBowl. */
    craftRecipe MakeCompanionDogsBowlFromPetDish
    {
        time = 60,
        timedAction = Making,
        category = Farming,
        Tags = InHandCraft;Farming,
        inputs
        {
            item 1 [Base.WaterDish] mode:destroy flags[ItemCount],
        }
        outputs
        {
            item 1 Base.CompanionDogsBowl,
        }
    }
}
